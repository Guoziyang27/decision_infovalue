108    def _test_overfit(self, signals: List[str], data: pd.DataFrame, training_ratio: float, overfit_tolerance: float, verbose: bool = False) -> bool:
109        """
110        Test if the model is overfitting.
111        """
112        training_data = data[:floor(len(data) * training_ratio)]
113        test_data = data[floor(len(data) * training_ratio):]
114        training_rational_payoff = _calculate_rational_payoff(signals, 
115                                                              training_data, training_data, 
116                                                              self.state, self.scoring_rule)
117        test_rational_payoff = _calculate_rational_payoff(signals, 
118                                                          training_data, test_data, 
119                                                          self.state, self.scoring_rule)
120        if verbose:
121            print(f"Training rational payoff: {training_rational_payoff}, Test rational payoff: {test_rational_payoff}, Overfit ratio: {(training_rational_payoff - test_rational_payoff) / abs(training_rational_payoff)}")
122        return (training_rational_payoff - test_rational_payoff) < overfit_tolerance * abs(training_rational_payoff)

124    def _find_opt_binning(self, signals: List[str], binning_method: str, overfit_tolerance: float, fit_test_ratio: float, verbose: bool = False) -> None:
125        """
126        Find the optimal binning for a feature.
127        """
128
129        use_data = self.dgp_data[signals + [self.state]].copy()
130        all_breaks = {feature: None for feature in signals}
131
132        while not self._test_overfit(signals, use_data, fit_test_ratio, overfit_tolerance, verbose):
133            if verbose:
134                print("Overfitting detected, finding optimal binning...")
135            feature_unique_values = use_data[signals].nunique()
136            max_unique_num_feature = signals[np.argmax(feature_unique_values)]
137            max_unique_num = feature_unique_values[max_unique_num_feature]
138            if binning_method == 'equal_probability':
139                _, breaks = pd.qcut(self.dgp_data[max_unique_num_feature], 
140                                    q=int(max_unique_num/2), 
141                                    retbins=True, 
142                                    duplicates='drop')
143            elif binning_method == 'equal_interval':
144                _, breaks = pd.cut(self.dgp_data[max_unique_num_feature], 
145                                   bins=int(max_unique_num/2), 
146                                   retbins=True, 
147                                   duplicates='drop', include_lowest=True)
148            use_data[max_unique_num_feature] = pd.cut(self.dgp_data[max_unique_num_feature], bins=breaks, labels=False, include_lowest=True)
149            all_breaks[max_unique_num_feature] = breaks
150        return use_data, all_breaks

322    def robust_analysis_on_v_shaped_scoring_rule(self, signals: List[str] | str, 
323                                                 compared_signals: List[str] | str,
324                                                base_signals: List[str] | str | None = None, 
325                                                ret_diff: bool = False,
326                                                grid_size: int = 100) -> float:
327        '''
328        Conduct a robust analysis on the V-shaped scoring rule.
329        '''
330        if base_signals is None:
331            base_signals = []
332        if isinstance(signals, str):
333            signals = [signals]
334        if isinstance(compared_signals, str):
335            compared_signals = [compared_signals]
336        if isinstance(base_signals, str):
337            base_signals = [base_signals]
338        if self.dgp_data[self.state].nunique() != 2:
339            raise ValueError(f"Robust analysis on the V-shaped scoring rule is only available for binary state")
340        if not all(signal in self.full_signals for signal in signals):
341            raise ValueError(f"Signals must be in the definition of full signals: {signals} not in {self.full_signals}")
342        if not all(signal in self.full_signals for signal in compared_signals):
343            raise ValueError(f"Compared signals must be in the definition of full signals: {compared_signals} not in {self.full_signals}")
344        if not all(signal in self.full_signals for signal in base_signals):
345            raise ValueError(f"Base signals must be in the definition of full signals: {base_signals} not in {self.full_signals}")
346        diff_list = []
347        for kink in np.linspace(0, 1, grid_size):
348            mscoring_rule = _define_v_shaped_scoring_rule(kink)
349            signal_aciv = _calculate_rational_payoff(signals + base_signals, self.all_use_data, self.all_use_data, self.state, mscoring_rule) - \
350            _calculate_rational_payoff(base_signals, self.all_use_data, self.all_use_data, self.state, mscoring_rule)
351            compared_signal_aciv = _calculate_rational_payoff(compared_signals + base_signals, self.all_use_data, self.all_use_data, self.state, mscoring_rule) - \
352            _calculate_rational_payoff(base_signals, self.all_use_data, self.all_use_data, self.state, mscoring_rule)
353            diff = signal_aciv - compared_signal_aciv
354            if not ret_diff:
355                if (diff < -1e-3):
356                    return False
357            else:
358                diff_list.append(diff)
359        if not ret_diff:
360            return True
361        else:
362            return np.array(diff_list), np.linspace(0, 1, grid_size)

373            signals = [signals]
374        if isinstance(base_signals, str):

386            combintation_signals.extend(list(combinations(left_signals, r)))
387        shapley_aciv = 0
388        # include_combination = [s for s in combintation_signals if set(signals).issubset(s) and not set(base_signals).issubset(s)]
389        for left_combination in combintation_signals:
390            exclude_combination = list(left_combination)
391            combination = exclude_combination + signals
392            aciv = _calculate_rational_payoff(combination + base_signals, self.all_use_data, self.all_use_data, self.state, self.scoring_rule) - \
393                _calculate_rational_payoff(exclude_combination + base_signals, self.all_use_data, self.all_use_data, self.state, self.scoring_rule)
394            shapley_aciv += aciv * math.factorial(len(exclude_combination)) * \
395                math.factorial(len(self.full_signals) - len(combination)) / \
396                    math.factorial(len(self.full_signals) - len(signals) + 1)
397        return shapley_aciv

109    def _test_overfit(self, signals: List[str], data: pd.DataFrame, training_ratio: float, overfit_tolerance: float, verbose: bool = False) -> bool:
110        """
111        Test if the model is overfitting.
112        """
113        training_data = data[:floor(len(data) * training_ratio)]
114        test_data = data[floor(len(data) * training_ratio):]
115        training_rational_payoff = _calculate_rational_payoff(signals, 
116                                                              training_data, training_data, 
117                                                              self.state, self.scoring_rule)
118        test_rational_payoff = _calculate_rational_payoff(signals, 
119                                                          training_data, test_data, 
120                                                          self.state, self.scoring_rule)
121        if verbose:
122            print(f"Training rational payoff: {training_rational_payoff}, Test rational payoff: {test_rational_payoff}, Overfit ratio: {(training_rational_payoff - test_rational_payoff) / abs(training_rational_payoff)}")
123        return (training_rational_payoff - test_rational_payoff) < overfit_tolerance * abs(training_rational_payoff)

125    def _find_opt_binning(self, signals: List[str], binning_method: str, overfit_tolerance: float, fit_test_ratio: float, verbose: bool = False) -> None:
126        """
127        Find the optimal binning for a feature.
128        """
129
130        use_data = self.dgp_data[signals + [self.state]].copy()
131        all_breaks = {feature: None for feature in signals}
132
133        while not self._test_overfit(signals, use_data, fit_test_ratio, overfit_tolerance, verbose):
134            if verbose:
135                print("Overfitting detected, finding optimal binning...")
136            feature_unique_values = use_data[signals].nunique()
137            max_unique_num_feature = signals[np.argmax(feature_unique_values)]
138            max_unique_num = feature_unique_values[max_unique_num_feature]
139            if binning_method == 'equal_probability':
140                _, breaks = pd.qcut(self.dgp_data[max_unique_num_feature], 
141                                    q=int(max_unique_num/2), 
142                                    retbins=True, 
143                                    duplicates='drop')
144            elif binning_method == 'equal_interval':
145                _, breaks = pd.cut(self.dgp_data[max_unique_num_feature], 
146                                   bins=int(max_unique_num/2), 
147                                   retbins=True, 
148                                   duplicates='drop', include_lowest=True)
149            use_data[max_unique_num_feature] = pd.cut(self.dgp_data[max_unique_num_feature], bins=breaks, labels=False, include_lowest=True)
150            all_breaks[max_unique_num_feature] = breaks
151        return use_data, all_breaks

324                                                 compared_signals: List[str] | str,

328        '''

330        '''

392            combination = exclude_combination + signals

397                    math.factorial(len(self.full_signals) - len(signals) + 1)
398        return shapley_aciv
